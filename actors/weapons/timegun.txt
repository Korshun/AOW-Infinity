		PSNG B 0 A_ChangeFlag ("PICKUP", 0)
		PSNG B 8 Bright A_RailAttack (0, 0, 1, "FF 96 00","FF D4 00", 0, 0, "GoldenPuff")
		PSNG B 0 A_ChangeFlag ("PICKUP", 1)
		PSNG A 35

		PSNG A 25 A_PlaySound ("80mm/reload", CHAN_WEAPON)
		goto Ready
	Fire.Teleport:
		PSNG C 27 BRIGHT A_PlayWeaponSound ("plasmacannon/fire")
		goto Nope
		PSNG C 0 A_JumpIfInventory ("TimeGunAmmo", 100, 1)
		goto OutOfAmmo
		PSNG C 0 A_TakeInventory ("TimeGunAmmo", 100)

		PSNG C 0 SetPlayerProperty (0, 1, 0)
		PSNG C 0 A_GiveInventory ("GoldHue", 1)

		PSNG A 0 A_SetTranslucent (1.0, 0)
		PSNG A 0 A_SpawnItemEx ("TimeGunFlareExplosion")
		PSNG A 0 SetPlayerProperty (0, 0, 0)
		PSNG C 0 A_TakeInventory ("IsTeleporting", 1)

		goto Ready

		PSNG C 0 A_PlayWeaponSound ("driver/fire")
		PSNG B 1 BRIGHT A_FireCustomMissile ("GoldenBall1Blue", random (-1.5, 1.5), 0, random (-2, 2), random (-2, 2))
		PSNG C 2
		PSNG C 0 A_PlayWeaponSound ("driver/fire")
		PSNG B 1 BRIGHT A_FireCustomMissile ("GoldenBall2Blue", random (-1.5, 1.5), 0, random (-2, 2), random (-2, 2))
		PSNG C 2
		PSNG C 0 A_Refire
	OutOfAmmo:
		PSNG A 2 A_PlaySound ("Plasma/FailFire", 6)
		PSNG A 0 A_PlaySound ("Plasma/Vapour", 7)
		PSNG A 0 A_TakeInventory ("TimeGunCharged", 1)
		PSNG AAAAAA 2 A_FireCustomMissile("OverheatSmokeSpawner",0,0,6,-2)
		PSNG A 35
		goto Ready
	AltFire:
		PSNG A 18 ACS_Execute (958, 0, 1)
		goto Ready
	}
