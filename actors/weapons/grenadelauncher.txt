// Grenade launcher
actor Grenade_Launcher : AOWWeapon replaces GrenadeLauncher 21046 {
	+WEAPON.NOAUTOAIM
	obituary "$OB_GRENADE" // "%o ate %k's grenade."
	inventory.pickupmessage "You got the grenade launcher!"
	weapon.selectionorder 12
	weapon.kickback 100
	weapon.ammotype "GrenadeAmmo"
	weapon.ammouse 1
	weapon.ammotype2 "GrenadeAmmo"
	weapon.ammouse2 1
	weapon.ammogive 2
	weapon.preferredskin "MarineRevenant"
	states {
	Spawn:
		GREN X -1
		stop
	Select:
		GREN A 1 A_Raise
		wait
	Deselect:
		GREN A 1 A_Lower
		wait
	Ready:
		GREN A 1 A_WeaponReady
		wait
	Fire:
		TNT1 A 0 A_JumpIfInventory("IsInSpawnRoom",1,"Ready")
		GREN A 6
		GREN A 5 A_PlayWeaponSound ("grenade/fire")
		GREN GH 1
		GREN I 0 A_GunFlash
		GREN I 0 A_JumpIfInventory("IsBlue",1,"FireBlue")
		GREN I 1 A_FireCustomMissile ("GrenadeGrenade", 0, 1, 12)
		GREN DEF 2
		GREN EDABC 1
		GREN A 0 A_ReFire
		goto Ready
	AltFire:
		TNT1 A 0 A_JumpIfInventory ("IsInSpawnRoom", 1, "Ready")
		GREN A 7
		GREN A 4 A_PlayWeaponSound ("grenade/fire")
		GREN GH 1
		GREN I 0 A_GunFlash
		GREN I 0 A_JumpIfInventory ("IsBlue",1,"AltFireBlue")
		GREN I 1 A_FireCustomMissile ("GrenadeBounce", 0, 1, 12)
		GREN DEF 2
		GREN EDABC 1
		GREN A 0 A_ReFire
		goto Ready
	FireBlue:
		GREN I 1 A_FireCustomMissile ("GrenadeGrenadeBlue", 0, 1, 12)
		GREN DEF 2
		GREN EDABC 1
		GREN A 0 A_ReFire
		goto Ready
	AltFireBlue:
		GREN I 1 A_FireCustomMissile ("GrenadeBounceBlue", 0, 1, 12)
		GREN DEF 2
		GREN EDABC 1
		GREN A 0 A_ReFire
		goto Ready
	}
}

actor GrenadeGrenade {
	PROJECTILE
	-RANDOMIZE
	-NOGRAVITY
	+LOWGRAVITY
	radius 11
	height 8
	speed 35
	damage (25)
	damagetype "RifleRed"
	seesound ""
	deathsound "weapons/rocklx"
	obituary "$OB_GRENADE" // "%o ate %k's grenade."
	decal "Scorch"
	states {
	Spawn:
		SGRN A 2 bright A_SpawnItemEx ("PulseBallSmokePuff", -25)
		loop
	Death:
		TNT1 B 0 bright A_NoGravity
		TNT1 B 0 bright A_Explode(55,128,1)
		TNT1 B 0 bright A_RadiusThrust(384,128,1) //Weeee
		TNT1 A 5 A_SpawnItemEx ("ExplosionMain")
		stop
	}
}

actor GrenadeBounce : GrenadeGrenade {
	+DOOMBOUNCE
	-LOWGRAVITY
	obituary "$OB_GRENADE_ALT" // "%o played catch with %k's grenade."
	damage (5)
	bouncecount 2
	states {
	Death:
		TNT1 A 0 bright A_NoGravity
		TNT1 A 0 bright A_Explode (40, 96, 1)
		TNT1 A 0 bright A_RadiusThrust(384,128,1) //Wee
		TNT1 A 5 A_SpawnItemEx ("ExplosionMain")
		stop
	}
}
