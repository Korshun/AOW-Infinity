// Frag grenade
actor FragGrenade : AOWWeapon {
	+NOALERT
	inventory.pickupmessage "Fragmentation Grenade"  
	inventory.pickupsound "misc/ammo_pkup"
	attacksound "frag/pinpull"
	weapon.kickback 5
	weapon.ammotype "FragGrenadeAmmo"
	weapon.ammouse 1
	weapon.ammogive 1
	weapon.ammotype2 "FragGrenadePower"
	weapon.ammouse2 0
	weapon.ammogive2 0
	scale 0.25
	Weapon.PreferredSkin "MarineFist"
	states {
	Ready:
		FRGS A 0 
		FRGS A 0 A_JumpIfNoAmmo(9)
		FRGS ABCDEFG 1
		FRGI A 1 A_WeaponReady
		goto Ready+9
		TNT1 A 1 A_WeaponReady
		loop
	Deselect:
		FRGI A 0 A_JumpIfNoAmmo(4)
		FRGI A 0 A_Lower
		FRGI A 1 A_Lower
		loop
		TNT1 A 0 A_Lower
		TNT1 A 1 A_Lower
		loop
	Select:
		FRGI A 0 A_Raise
		loop
	Fire:
		FRGI A 2 A_TakeInventory ("FragGrenadePower", 100)
		FRG1 BCDEFGH 2
		FRG1 I 0 A_PlayWeaponSound ("frag/pinpull")
		FRG1 IJK 1
		TNT1 A 10
		TNT1 A 1 A_Refire
		TNT1 A 3 A_PlaySound ("frag/voice")
		FRG2 AB 2
		FRG2 C 2 A_PlayWeaponSound ("frag/throw")
		TNT1 A 0 A_TakeInventory ("FragGrenadePower", 100)
		FRGI A 0 A_JumpIfInventory ("IsBlue", 1, "FireBlue")
		FRG2 D 2 A_FireCustomMissile ("FragGrenadeThrown1",-1,1,10,5)
		FRG2 EFG 2
		goto Ready
	Hold:
		TNT1 A 4 A_GiveInventory("FragGrenadePower", 6)
		TNT1 A 1 A_ReFire
		TNT1 A 3 A_PlaySound("frag/voice")
		FRGI A 0 A_JumpIfInventory("IsBlue",1,"HoldBlue")
		TNT1 A 0 A_JumpIfInventory("FragGrenadePower",80,"Throw5")
		TNT1 A 0 A_JumpIfInventory("FragGrenadePower",60,"Throw4")
		TNT1 A 0 A_JumpIfInventory("FragGrenadePower",40,"Throw3")
		TNT1 A 0 A_JumpIfInventory("FragGrenadePower",20,"Throw2")
		FRG2 AB 2
		FRG2 C 2 A_PlayWeaponSound("frag/throw")
		TNT1 A 0 A_TakeInventory("FragGrenadePower",100)
		FRG2 D 2 A_FireCustomMissile("FragGrenadeThrown1",-1,1,10,5)
		FRG2 EFG 2
		goto Ready
	Throw2:
		FRG2 AB 2
		FRG2 C 2 A_PlayWeaponSound("frag/throw")
		TNT1 A 0 A_TakeInventory("FragGrenadePower",100)
		FRG2 D 2 A_FireCustomMissile("FragGrenadeThrown2",-1,1,10,5)
		FRG2 EFG 2
		goto Ready
	Throw3:
		TNT1 A 2
		TNT1 A 2 A_PlayWeaponSound("frag/throw")
		TNT1 A 0 A_TakeInventory("FragGrenadePower",100)
		FRG3 A 2 A_FireCustomMissile("FragGrenadeThrown3",-1,1,10,15)
		FRG3 BCD 2
		TNT1 AA 2
		goto Ready
	Throw4:
		TNT1 A 2
		TNT1 A 2 A_PlayWeaponSound("frag/throw")
		TNT1 A 0 A_TakeInventory("FragGrenadePower",100)
		FRG3 A 2 A_FireCustomMissile("FragGrenadeThrown4",-1,1,10,15)
		FRG3 BCD 2
		TNT1 AA 2
		goto Ready
	Throw5:
		TNT1 A 2
		TNT1 A 2 A_PlayWeaponSound("frag/throw")
		TNT1 A 0 A_TakeInventory("FragGrenadePower",100)
		FRG3 A 2 A_FireCustomMissile("FragGrenadeThrown5",-1,1,10,15)
		FRG3 BCD 2
		TNT1 AA 2
		goto Ready
	FireBlue:
		FRG2 D 2 A_FireCustomMissile("FragGrenadeThrown1Blue",-1,1,10,5)
		FRG2 EFG 2
		goto Ready
	HoldBlue:
		TNT1 A 0 A_JumpIfInventory("FragGrenadePower",80,"Throw5Blue")
		TNT1 A 0 A_JumpIfInventory("FragGrenadePower",60,"Throw4Blue")
		TNT1 A 0 A_JumpIfInventory("FragGrenadePower",40,"Throw3Blue")
		TNT1 A 0 A_JumpIfInventory("FragGrenadePower",20,"Throw2Blue")
		FRG2 AB 2
		FRG2 C 2 A_PlayWeaponSound("frag/throw")
		TNT1 A 0 A_TakeInventory("FragGrenadePower",100)
		FRG2 D 2 A_FireCustomMissile("FragGrenadeThrown1Blue",-1,1,10,5)
		FRG2 EFG 2
		goto Ready
	Throw2Blue:
		FRG2 AB 2
		FRG2 C 2 A_PlayWeaponSound("frag/throw")
		TNT1 A 0 A_TakeInventory("FragGrenadePower",100)
		FRG2 D 2 A_FireCustomMissile("FragGrenadeThrown2Blue",-1,1,10,5)
		FRG2 EFG 2
		goto Ready
	Throw3Blue:
		TNT1 A 2
		TNT1 A 2 A_PlayWeaponSound("frag/throw")
		TNT1 A 0 A_TakeInventory("FragGrenadePower",100)
		FRG3 A 2 A_FireCustomMissile("FragGrenadeThrown3Blue",-1,1,10,15)
		FRG3 BCD 2
		TNT1 AA 2
		goto Ready
	Throw4Blue:
		TNT1 A 2
		TNT1 A 2 A_PlayWeaponSound("frag/throw")
		TNT1 A 0 A_TakeInventory("FragGrenadePower",100)
		FRG3 A 2 A_FireCustomMissile("FragGrenadeThrown4Blue",-1,1,10,15)
		FRG3 BCD 2
		TNT1 AA 2
		goto Ready
	Throw5Blue:
		TNT1 A 2
		TNT1 A 2 A_PlayWeaponSound("frag/throw")
		TNT1 A 0 A_TakeInventory("FragGrenadePower",100)
		FRG3 A 2 A_FireCustomMissile("FragGrenadeThrown5Blue",-1,1,10,15)
		FRG3 BCD 2
		TNT1 AA 2
		goto Ready
	Spawn:
		TNT1 A 0
		TNT1 A 0 A_SpawnItem ("FragGrenadeAmmo")
		stop
	}
}

actor FragGrenadeAmmo : Ammo {
	inventory.pickupmessage "Picked up a frag grenade."
	inventory.amount 1
	inventory.maxamount 1
	ammo.backpackamount 1
	ammo.backpackmaxamount 1
	scale 0.25
	states {
	Spawn:
		FRGP A -1
		stop
	}
}

actor FragGrenadePower : Ammo {
	+IGNORESKILL
	inventory.amount 0
	inventory.maxamount 100
	ammo.backpackamount 0
	ammo.backpackmaxamount 100
}

actor FragGrenadeThrown1 {
	Obituary "%o didn't see %k's frag grenade."
	radius 5
	height 5
	speed 17
	damage (0)
	seesound ""
	deathsound "weapons/rocklx"
	PROJECTILE
	-NOGRAVITY
	+RANDOMIZE
	+HEXENBOUNCE
	+NOEXTREMEDEATH
	+FORCERADIUSDMG
	+CANBOUNCEWATER
	+BOUNCEONACTORS
	Scale 0.22
	gravity 0.6
	bouncefactor 0.5
	damagetype "RifleRed"
	reactiontime 1
	states {
	Spawn:
		FRGX ABCDEFGHA 5
		FRGX BB 5
		FRGX BB 0 A_Countdown
	Death:
		TNT1 B 0 bright A_NoGravity
		TNT1 A 0 bright A_StopSoundEx("Body")
		TNT1 A 5 A_Explode (190, 256)
		TNT1 A 1 A_SpawnItemEx("FlameDecay", random(-16,16), random(-16,16), 4, 0, 0, 0, 0, 0, 160)
		TNT1 AAAAAAAA 0 A_CustomMissile("FragGrenadeShrapnel",0,0,random(0,360),6,random(2,80))
		TNT1 A 5 A_SpawnItem ("ExplosionMain")
		stop
	}
}

actor FragGrenadeShrapnel {
	PROJECTILE
	+RIPPER
	+BLOODSPLATTER
	+HEXENBOUNCE
	Radius 10
	Height 8
	Damage (4)
	Speed 60
	Obituary "%o was shredded by %k's shrapnel."
	Scale 0.3
	Gravity 0.6
	bouncecount 10
	Damagetype "RifleRed"
	states
	{
	Spawn:
		SHRA A 5 BRIGHT
		SHRA A 4 A_Gravity
		SHRA A 55
		goto Death1
	Death1:
		SHRA A 4 A_FadeOut(0.2)
		loop
	}
}
