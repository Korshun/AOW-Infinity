actor TiberiumAutorifleWeak : AOWWeapon {
	+AMMO_OPTIONAL
	inventory.pickupmessage "You got the Tiberium Autorifle!"
	weapon.selectionorder 13
	weapon.kickback 100
	weapon.ammotype "TE21Burst"
	weapon.ammouse 1
	weapon.ammogive 3
	weapon.ammotype2 "TE21CalAmmo"
	weapon.ammouse2 1
	weapon.ammogive2 20
	obituary "%o was skewered by %k's Tiberium Autorifle."
	Weapon.PreferredSkin "MarineMachineGun"
	states
	{
	Ready:
		E21C A 1 A_WeaponReady
		loop
	Deselect:
		E21C A 1 A_Lower
		loop
	Select:
		E21C A 1 A_Raise
		loop
	Fire:
		NULL A 0 A_JumpIfInventory ("IsInSpawnRoom", 1, "Ready")
		NULL A 0 A_JumpIfNoAmmo ("Reload")
		NULL A 0 A_GunFlash
		NULL A 0 A_PlaySound("toxin/fire")
		NULL A 0 A_JumpIfInventory ("IsBlue",1,"Fire")
		E21C B 2 bright A_FireCustomMissile ("TiberiumDart", 0, 1, 10)
		E21C C 2 bright
		E21C D 2 A_JumpIfNoAmmo ("Reload")
		E21C A 2 A_ReFire
		goto Ready
	Fire:
		E21C B 2 bright A_FireCustomMissile ("TiberiumDartBlue", 0, 1, 10)
		E21C C 2 bright
		E21C D 2 A_JumpIfNoAmmo ("Reload")
		E21C A 2 A_ReFire
		goto Ready
	Reload:
		NULL A 0 A_JumpIfInventory ("TE21CalAmmo", 3, 2)
		E21C A 5 A_CheckReload
		goto Ready
		E21C A 15
		NULL A 0 A_TakeInventory ("TE21CalAmmo", 3)
		NULL A 0 A_GiveInventory ("TE21Burst", 3)
		goto Ready
	Flash:
		NULL A 2 A_Light1
		NULL A 2 A_Light2
		NULL A 0 A_Light0
		stop
	Spawn:
		E21C X -1
		stop
	}
}
