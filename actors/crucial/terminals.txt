actor BaseTerminal {
	damagefactor "FistRed", 0.0
	damagefactor "FistBlue", 0.2
	damagefactor "BulletRed", 0.0
	damagefactor "BulletBlue", 0.5
	damagefactor "RifleRed", 0.0
	damagefactor "RifleBlue", 0.5
	damagefactor "ExplosionRed", 0.0
	damagefactor "ExplosionBlue", 0.75
	damagefactor "FireRed", 0.0
	damagefactor "FireBlue", 0.6
	damagefactor "FireSpecial", 0.0
	damagefactor "ChemicalRed", 0.0
	damagefactor "ChemicalBlue", 0.75
	damagefactor "ChemicalSpecial", 0.0
	damagefactor "LaserRed", 0.0
	damagefactor "LaserBlue", 1.0
	damagefactor "RazorWireRed", 0.0
	damagefactor "RazorWireBlue", 0.0
	damagefactor "ExplosionMechRed", 0.0
	damagefactor "ExplosionMechBlue", 0.25
	damagefactor "TimeRed", 0.0
	damagefactor "TimeBlue", 1.0
	damagefactor "Disarm", 0.0
	damagefactor "Flash", 0.0
	damagefactor "Superweapon", 0.0
	damagefactor "SuperweaponIon", 1.0
	damagefactor "Slime", 0.0 //Mines
	damagefactor "MassDriverRed", 0.0
	damagefactor "MassDriverBlue", 0.25
	damagefactor "Mechstomp", 0.0
	damagefactor "Monster", 0.0
	damagefactor "Deconstruction", 0.0
	height 120
	mass 0x7FFFFFFF
	speed 100 // used for base hud health percentages! DON'T TOUCH IT
	+FULLVOLACTIVE
	-SOLID
	+SHOOTABLE
	+NOBLOOD
	+DONTGIB
	+NOICEDEATH
	+QUICKTORETALIATE
	+NEVERRESPAWN
	+GHOST
	
	damagetype "ExplosionRed"
	obituary "%o got killed with a \cGRed\c- building."
	radius 16
	deathsound "grenade/explode"
	painchance 255
	
	states {
	Spawn:
		TNT1 A 0
		TNT1 A 0 A_GiveInventory ("IsMCT", 1)    // Needed for turrets in exp pack
		TNT1 A -1
		loop
	Idle:
		TNT1 A -1
		stop
	SpawnDone:
		TNT1 A 0
		TNT1 A 0 A_ChangeFlag("ISMONSTER",1)
		TNT1 A 0 A_ClearTarget
		TNT1 A -1 A_Stop
		stop
	Death.Disarm:
		TNT1 A 0
		TNT1 A 0 A_Print("*** FLUKE! *** Red Terminal was destroyed by disarm")
		goto Death
	Death.MechStomp:
		TNT1 A 0
		TNT1 A 0 A_Print("*** FLUKE! *** Red Terminal was destroyed by mechstomp")
		goto Death
	Pain:
		goto Spawn
	Pain.Chemical:
	Pain.Fire:
		goto Pain+1
	Death:
		TNT1 A 5 A_Scream
		TNT1 A 5 A_Scream
		TNT1 A 0 bright A_PlayWeaponSound("weapons/rocklx")
		TNT1 A 0 A_SpawnItemEx ("TerminalExplosion")
		TNT1 A 0 A_SpawnItemEx("ExplosionMainLarge", 0, 0, 0, 0, 0, 0, 0, 128, 0)
		TNT1 A 5 A_Scream
		TNT1 A 0 bright A_Scream
		TNT1 A 0 A_SpawnItemEx ("TerminalExplosion")
		TNT1 A 0 bright A_PlayWeaponSound("weapons/rocklx")
		TNT1 A 0 A_SpawnItemEx ("TerminalExplosion")
		TNT1 A 0 A_SpawnItemEx("ExplosionMainLarge", 0, 0, 0, 0, 0, 0, 0, 128, 0)
		TNT1 A 5 A_Scream
		TNT1 A 5 A_Scream
		TNT1 A 0 bright A_PlayWeaponSound("weapons/rocklx")
		TNT1 A 0 A_SpawnItemEx ("TerminalExplosion")
		TNT1 A 0 A_SpawnItemEx("ExplosionMainLarge", 0, 0, 0, 0, 0, 0, 0, 128, 0)
		TNT1 A 5 A_Scream
		TNT1 A 5 A_Scream
		TNT1 A 0 bright A_Scream
		TNT1 A 5 
		TNT1 A -1 A_Scream
		stop
	}
}
