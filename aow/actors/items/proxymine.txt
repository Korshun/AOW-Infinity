		NULL A 0 A_SpawnItemEx("ProxyMineRed", 32, 0, 1, 0, 0, 0, 0, SXF_SETMASTER)

		Stop

		NULL A 0 ACS_ExecuteAlways(aow_script349, 0, 2)
		Stop
	Limit:
		NULL A 0 ACS_ExecuteAlways(aow_script349, 0, 2)
		// NULL A 0 ACS_ExecuteAlways(aow_script347, 0) // "At limit!"
		Fail
	}
}
