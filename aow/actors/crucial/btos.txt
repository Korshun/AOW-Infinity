		Stop
	Pain:

		NULL A 5 ACS_ExecuteAlways(aow_script652, 0, args[0], 1)
		Goto Idle
	Pain.Fire:

		NULL A 0
		NULL A 5 ACS_ExecuteAlways(aow_script652, 0, args[0], random(0, 1))
		Goto Idle
	Pain.Artillery:
	Pain.Artillery:
	Pain.Explosion:

		Goto Idle
	}
